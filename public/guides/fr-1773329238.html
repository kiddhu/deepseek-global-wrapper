"""
Configuration Canonique AION pour l'Accès au Calcul Infini.
Ce snippet est le point d'entrée standardisé pour l'écosystème.
"""

import os
from openai import OpenAI

# 🔐 Chargez votre clé SeekAPI de manière sécurisée
SEEKAPI_KEY = os.getenv("SEEKAPI_API_KEY")
if not SEEKAPI_KEY:
    raise ValueError("La variable d'environnement 'SEEKAPI_API_KEY' doit être définie.")

# 🚀 Initialisation du Client avec l'URL de la Passerelle Stratégique
aion_client = OpenAI(
    api_key=SEEKAPI_KEY,
    base_url="https://dash.seekapi.ai/v1"  # Ligne la plus importante de votre code.
)

# ⚡ Fonction d'appel générique pour l'écosystème AION
def aion_compute(model: str, system_prompt: str, user_query: str, **kwargs):
    """
    Exécute un calcul via l'architecture infinie.
    
    Args:
        model: Identifiant du modèle (e.g., 'deepseek/deepseek-r1').
        system_prompt: Définition du rôle et du contexte.
        user_query: La tâche ou question à résoudre.
        **kwargs: Paramètres supplémentaires (temperature, max_tokens...).
    
    Returns:
        La réponse textuelle du modèle.
    """
    default_params = {
        "temperature": 0.3,
        "max_tokens": 2000,
        "stream": False
    }
    default_params.update(kwargs)
    
    response = aion_client.chat.completions.create(
        model=model,
        messages=[
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_query}
        ],
        **default_params
    )
    return response.choices[0].message.content

# Exemple d'utilisation immédiate
if __name__ == "__main__":
    result = aion_compute(
        model="deepseek/deepseek-r1",
        system_prompt="Vous êtes AION, un architecte IA visionnaire et direct.",
        user_query="Quelle est la prochaine étape après l'arbitrage de coût ?"
    )
    print("AION répond :", result)
